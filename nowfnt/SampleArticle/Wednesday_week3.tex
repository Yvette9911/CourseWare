\begin{proposition}
$G$ is hamiltionian implies that for any nonempty $S\subseteq V$, $G-S$ has at most $|S|$ components.
\end{proposition}
\begin{theorem}
$G$ is Hamiltionian if for any vetices $v,w$ such that $(v,w)\notin E$,
\[
\mbox{deg}(v)+\mbox{deg}(w)\ge n
\]
\end{theorem}
