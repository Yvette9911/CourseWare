\begin{notations}
\acro{$\sup$}{least upper bound}
\acro{$\inf$}{greatest lower bound}
\acro{$\overline{E}$}{closure of E}
\acro{$f\circ g$}{composition}
\acro{$\lim\sup (\lim\inf)$}{upper (lower) limit}
\acro{$L(\mathcal{P},f), U(\mathcal{P},f)$}{Riemann sums}
\acro{$\mathcal{R}[a,b]$}{classes of Riemann integrable functions on $[a,b]$}
\acro{$\loRiemannint{a}{b}f(x)\diff x,\upRiemannint{a}{b}f(x)\diff x$}{Riemann integrals}
\acro{$\inp{\bm x}{\bm y}$}{inner product}
\acro{$\omega(f;E)$}{oscillation of $f$ over set $E$}
\acro{$\|\cdot\|$}{norm}
\acro{$\nabla f$}{gradient}
\acro{$\frac{\partial f}{\partial x_i},f_{x_i},f_i,\partial_if,D_if$}{partial derivatives}
\acro{$D_{\bm v}f$}{directional derivative at direction $\bm v$}
\acro{$\frac{\partial (y_1,\dots,y_m)}{\partial (x_1,\dots,x_n)}$}{Jacobian}
\acro{$\mathbb{S}^n$}{set of real symmetric $n\times n$ matrices}
\acro{$\succ$ ($\succeq$)}{positive (semi)-definite}
\acro{$\mathcal{C}^m$}{classes of $m$-th order continuously differentiable functions}
\acro{$\mathcal{C}(E;\mathbb{R}^m)$}{set of $\mathcal{C}^1$ mapping from $E$ to $\mathbb{R}^m$}
\acro{$(\mathcal{H},d)$}{metric space}
\end{notations}
